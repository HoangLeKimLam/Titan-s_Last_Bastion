from dataclasses import dataclass, field, asdict
import json


@dataclass
class ResourceBundle:
    """Túi tài nguyên. Hỗ trợ +, *, >= để code gọn."""

    wood:             int = 0
    stone:            int = 0
    gas:              int = 0
    food:             int = 0
    ore:              int = 0
    crystal:          int = 0
    serum:            int = 0
    anti_armor_bolt:  int = 0

    def __add__(self, other: "ResourceBundle") -> "ResourceBundle":
        """r1 + r2 — cộng 2 túi lại (khi loot về kho)."""
        return ResourceBundle(
            wood            = self.wood            + other.wood,
            stone           = self.stone           + other.stone,
            gas             = self.gas             + other.gas,
            food            = self.food            + other.food,
            ore             = self.ore             + other.ore,
            crystal         = self.crystal         + other.crystal,
            serum           = self.serum           + other.serum,
            anti_armor_bolt = self.anti_armor_bolt + other.anti_armor_bolt,
        )

    def __mul__(self, factor: float) -> "ResourceBundle":
        """r * 0.8 — áp penalty mất 20% khi thua màn."""
        return ResourceBundle(
            wood  = int(self.wood  * factor),
            stone = int(self.stone * factor),
            gas   = int(self.gas   * factor),
            food  = int(self.food  * factor),
        )

    def __ge__(self, cost: "ResourceBundle") -> bool:
        """r >= cost — kiểm tra đủ tài nguyên để xây/upgrade."""
        return (
            self.wood            >= cost.wood
            and self.stone       >= cost.stone
            and self.gas         >= cost.gas
            and self.food        >= cost.food
            and self.ore         >= cost.ore
            and self.crystal     >= cost.crystal
            and self.serum       >= cost.serum
        )

# core/event_bus.py
from typing import Callable, Dict, List


class GameEventBus:
    """Singleton Observer — phát và lắng nghe sự kiện.
    
    Publisher chỉ cần publish('tên_sự_kiện', data).
    Subscriber tự đăng ký và tự phản ứng.
    Các class không cần biết nhau.
    """

    _instance = None

    def __init__(self):
        self._listeners: Dict[str, List[Callable]] = {}

    @classmethod
    def get_instance(cls) -> "GameEventBus":
        if cls._instance is None:
            cls._instance = GameEventBus()
        return cls._instance

    def subscribe(self, event: str, callback: Callable):
        """Đăng ký lắng nghe sự kiện.
        
        Ví dụ:
            bus.subscribe('wall_breached', hud.on_wall_breached)
            bus.subscribe('titan_died',    resource_manager.on_titan_died)
        """
        if event not in self._listeners:
            self._listeners[event] = []
        self._listeners[event].append(callback)

    def publish(self, event: str, data=None):
        """Phát sự kiện đến tất cả subscriber.
        
        Ví dụ:
            bus.publish('wall_breached', {'section': self})
            bus.publish('titan_died',    {'titan': self, 'reward': bundle})
        """
        for callback in self._listeners.get(event, []):
            callback(data)

class MeleeRushStrategy(TitanAttackStrategy):
    """Lao vào đánh cận chiến — hệ số và dtype cấu hình được.
    Nhóm dùng: Titan cơ bản, Titan tốc độ cao, berserk mode.
    Vd: MeleeRushStrategy()               → damage thường
        MeleeRushStrategy(damage_mult=2.0) → switch runtime khi berserk

        self._dtype       = dtype

            amount=int(attacker._damage * self._damage_mult),
            dtype=self._dtype

class Titan(Entity, IAttackable, IMovable):
    """Cha trừu tượng của mọi Titan.
    
    Không tạo trực tiếp — chỉ để kế thừa.
    Chứa AI logic chung: tìm target, di chuyển, tấn công.
    """

    def __init__(self, x: float, y: float, config: dict):
        super().__init__(x, y)
        self._hp               = config['hp']
        self._max_hp           = config['hp']
        self._speed            = config['speed']
        self._damage           = config['damage']
        self._reward           = config.get('reward', {})
        self._target           = None
        self._attack_strategy  = None   # gán ở class con
        self._attack_range     = config.get('attack_range', 60)
        self._attack_cooldown  = config.get('attack_cooldown', 1.5)
        self._attack_timer     = 0.0

    # ─── METHODS BẮT BUỘC (Entity yêu cầu) ─────────────

    def update(self, dt: float):
        """Chạy mỗi frame: tìm target → di chuyển → tấn công."""
        if not self.is_alive:
            return

        self._attack_timer -= dt

        # Tìm target mới nếu cần
        if self._target is None or not self._target.is_alive:
            self._target = self._find_best_target()

        # Di chuyển về phía target
        if self._target is not None:
            dist = self._distance_to(self._target)

            if dist > self._attack_range:
                self._move_toward(self._target, dt)
            elif self._attack_timer <= 0:
                self._attack_strategy.execute(self, self._target)
                self._attack_timer = self._attack_cooldown

    def draw(self, screen):
        """Vẽ sprite + HP bar."""
        pass   # pygame.draw.rect(...)

    # ─── IAttackable ─────────────────────────────────────

    def take_damage(self, amount: int, dtype: str):
        """Nhận damage. Class con override nếu có giáp."""
        self._hp -= amount
        if self._hp <= 0:
            self.on_death()

    def on_death(self):
        """Gọi khi HP = 0. Emit event để ResourceManager cộng thưởng."""
        self.is_alive = False
        GameEventBus.get_instance().publish(
            'titan_died',
            {'titan': self, 'reward': self._reward}
        )

    # ─── IMovable ─────────────────────────────────────────

    def move(self, destination: tuple):
        """Di chuyển trực tiếp đến tọa độ (dùng cho teleport)."""
        self.x, self.y = destination

    # ─── PRIVATE HELPERS ─────────────────────────────────

    def _find_best_target(self):
        """Tìm mục tiêu theo thứ tự ưu tiên.
        
        Ưu tiên 1: HQ nếu có thể vào thẳng
        Ưu tiên 2: WallSection cản đường
        Ưu tiên 3: Tower/Soldier đang tấn công mình
        Fallback : HQ
        """
        from systems.world_query import WorldQuery
        hq = WorldQuery.get_headquarters()

        if WorldQuery.can_reach_direct(self, hq):
            return hq

        wall = WorldQuery.find_blocking_wall(self, hq)
        if wall:
            return wall

        attacker = WorldQuery.find_nearest_attacker(self)
        if attacker:
            return attacker

        return hq

    def _move_toward(self, target, dt: float):
        """Di chuyển về phía target với tốc độ _speed * dt."""
        dx = target.x - self.x
        dy = target.y - self.y
        dist = (dx**2 + dy**2) ** 0.5
        if dist > 0:
            self.x += (dx / dist) * self._speed * dt
            self.y += (dy / dist) * self._speed * dt

    def _distance_to(self, target) -> float:
        """Khoảng cách Euclidean đến target."""
        dx = target.x - self.x
        dy = target.y - self.y
        return (dx**2 + dy**2) ** 0.5

class RegularTitan(Titan):
    """Titan cơ bản — đấm thẳng vào tường. Berserk khi HP < 30%."""

    def __init__(self, x: float, y: float, config: dict):
        super().__init__(x, y, config)
        self._attack_strategy = MeleeRushStrategy()
        self._berserk         = False

    def update(self, dt: float):
        super().update(dt)
        # Berserk: HP < 30% → speed tăng + switch sang đòn đánh mạnh hơn
        if not self._berserk and self._hp / self._max_hp < 0.3:
            self._berserk = True
            self._speed *= 1.5
            self._attack_strategy = MeleeRushStrategy(damage_mult=2.0)


class ArmoredTitan(Titan):
    """Titan giáp — chặn 60% damage thường. Override take_damage()."""

    ARMOR_REDUCTION = 0.6   # chặn 60%

    def __init__(self, x: float, y: float, config: dict):
        super().__init__(x, y, config)
        self._attack_strategy = ArmoredRamStrategy()
        self._armor_intact    = True

    def take_damage(self, amount: int, dtype: str):
        """Override: kiểm tra giáp trước khi trừ HP."""
        if self._armor_intact:
            if dtype == 'anti_armor':
                # Anti-Armor Bolt xuyên giáp hoàn toàn
                actual = amount
                self._armor_intact = False   # giáp vỡ!
                # Switch strategy khi giáp vỡ
                self._attack_strategy = HeavyStrikeStrategy()
            else:
                # Đạn thường chỉ gây 40%
                actual = int(amount * (1 - self.ARMOR_REDUCTION))
        else:
            actual = amount

        self._hp -= actual
        if self._hp <= 0:
            self.on_death()


class CrawlerTitan(Titan):
    """Titan leo tường — dùng dây bypass tháp mặt đất."""

class ODMSwingStrategy(SoldierAttackStrategy):
    """Scout du dây — xác suất chọn hành động dựa trên Gas% / HP%.
    3 lựa chọn: Swing Next (bảo thủ), Land & Stay (cân bằng), Land & Fight (mạo hiểm).
    Cấu hình: gas_coeff, hp_coeff, level_coeff → điều chỉnh xác suất.
    """
    def __init__(self, gas_coeff: float = 0.6, hp_coeff: float = 0.3, level_coeff: float = 0.1):
        self._gas_coeff   = gas_coeff
        self._hp_coeff    = hp_coeff
        self._level_coeff = level_coeff

    def _choose_action(self, soldier) -> str:
        """Chọn hành động dựa xác suất (TBD: điều chỉnh coefficients)."""
        gas_pct  = soldier._weapon._ammo / soldier._weapon._max_ammo
        hp_pct   = soldier._hp / soldier._max_hp
        level    = getattr(soldier, '_level', 1)  # if not exist, default 1

        # Xác suất: Swing (bảo thủ) cao khi gas thấp
        prob_swing = gas_pct * self._gas_coeff
        # Xác suất: Land&Stay (cân bằng) nếu HP trung bình
        prob_stay  = (hp_pct * self._hp_coeff) + 0.3
        # Xác suất: Land&Fight (mạo hiểm) nếu HP cao + level cao
        prob_fight = ((hp_pct + level / 10) * self._level_coeff) + 0.1

        # Chuẩn hóa
        total = prob_swing + prob_stay + prob_fight
        prob_swing /= total
        prob_stay  /= total
        prob_fight /= total

        r = random.random()
        if r < prob_swing:
            return 'swing_next'
        elif r < prob_swing + prob_stay:
            return 'land_stay'
        else:
            return 'land_fight'

    def execute(self, soldier, target: IAttackable):
        if not soldier._weapon or not soldier._weapon.has_ammo():
            return

        action = self._choose_action(soldier)
        if action == 'swing_next':
            soldier._weapon.use(target)   # chém rồi đu tiếp
        elif action == 'land_stay':
            soldier._weapon.use(target)   # hạ cánh, chém vài lần
        else:  # land_fight
            soldier._weapon.use(target)   # hạ cánh, chém đến chết

    def should_retreat(self, soldier) -> bool:
        # Scout rút lui nếu hết gas HOẶC HP < 20%
        return ((self._weapon and not soldier._weapon.has_ammo()) or
                soldier._hp / soldier._max_hp < 0.2)

class Soldier(Entity, IAttackable, IMovable):
    """Cha của mọi lính. Lính đóng quân tại tháp được giao."""

    def __init__(self, x: float, y: float):
        super().__init__(x, y)
        self._hp            = 0     # class con gán
        self._max_hp        = 0
        self._attack_range  = 0
        self._attack_timer  = 0.0
        self._attack_cooldown = 1.0
        self._target        = None
        self._assigned_tower = None  # tháp được giao
        self._home_pos      = (x, y)  # vị trí gốc tại tháp
        self._weapon        = None   # HAS-A weapon
        self._state         = 'idle' # idle|combat|retreat|reload

    def update(self, dt: float):
        """Logic mỗi frame theo state."""
        if not self.is_alive:
            return

        self._attack_timer -= dt

        if self._state == 'retreat':
            self._update_retreat(dt)
        elif self._state == 'reload':
            self._update_reload(dt)
        elif self._state == 'combat':
            self._update_combat(dt)
        else:
            self._update_idle(dt)

        self._check_state_transition()

    def take_damage(self, amount: int, dtype: str):
        self._hp -= amount
        if self._hp <= 0:
            self.on_death()

    def on_death(self):
        self.is_alive = False
        GameEventBus.get_instance().publish(
            'soldier_died', {'soldier': self}
        )

    def move(self, destination: tuple):
        self.x, self.y = destination

    def get_hp_color(self) -> tuple:
        """Trả về màu HP bar. HUD gọi để vẽ."""
        ratio = self._hp / self._max_hp
        if   ratio > 0.6: return (80,  200, 120)  # xanh
        elif ratio > 0.3: return (240, 190, 50)   # vàng
        elif ratio > 0.1: return (220, 70,  70)   # đỏ
        else:             return (100, 100, 100)  # xám

    def assign_to_tower(self, tower):
        """Điều phối lính đến tháp. Lính lấy vị trí home từ tháp."""
        self._assigned_tower = tower
        self._home_pos = tower.get_garrison_pos()

    def draw(self, screen):
        pass

    # ─── PRIVATE STATE UPDATES ───────────────────────────

    def _update_combat(self, dt: float):
        """Tìm target và tấn công trong tầm."""
        if self._target is None or not self._target.is_alive:
            self._target = self._find_target_in_range()

        if (self._target is not None
                and self._attack_timer <= 0
                and self._weapon is not None):
            self._weapon.use(self._target)
            self._attack_timer = self._attack_cooldown

    def _update_retreat(self, dt: float):
        """Di chuyển về tháp home để hồi máu."""
        hx, hy = self._home_pos
        dx, dy = hx - self.x, hy - self.y
        dist = (dx**2 + dy**2) ** 0.5
        if dist < 10:
            self._state = 'reload'   # đến nơi → reload
        else:
            speed = 80
            self.x += (dx / dist) * speed * dt
            self.y += (dy / dist) * speed * dt

    def _update_reload(self, dt: float):
        """Đứng tại tháp hồi máu và nạp vũ khí."""
        heal_rate = 30   # HP/giây
        self._hp = min(self._max_hp, self._hp + heal_rate * dt)
        if self._weapon:
            self._weapon.reload()
        # Khi đủ máu và đạn → quay lại chiến đấu
        if (self._hp >= self._max_hp * 0.6
                and (self._weapon is None or self._weapon.has_ammo())):
            self._state = 'combat'

    def _update_idle(self, dt: float):
        """Đứng yên tại vị trí home, quét radar."""
        target = self._find_target_in_range()
        if target:
            self._state = 'combat'

    def _check_state_transition(self):
        """Kiểm tra chuyển state tự động theo HP và ammo."""
        hp_ratio = self._hp / self._max_hp
        no_ammo  = self._weapon is not None and not self._weapon.has_ammo()

        if hp_ratio <= 0.3 or no_ammo:
            if self._state not in ('retreat', 'reload'):
                self._state  = 'retreat'
                self._target = None

    def _find_target_in_range(self):
        from systems.world_query import WorldQuery
        return WorldQuery.find_nearest(
            cx=self.x, cy=self.y,
            entity_type='titan',
            max_range=self._attack_range
        )

class Wall(Entity, IAttackable):
    """Composite — chứa nhiều WallSection.

    Gọi wall.take_damage(100, pos=(x,y)) →
    Wall tự tìm section gần pos nhất → trừ HP section đó.
    Bên ngoài không cần biết bên trong có bao nhiêu section.
    """

    def __init__(self, name: str, positions: list):
        """
        Tham số:
            name:      str   — 'maria', 'rose', 'sina'
            positions: list  — danh sách (x, y) của từng section
        """
        super().__init__(x=0, y=0)
        self.name      = name
        self._sections = [
            WallSection(px, py) for px, py in positions
        ]

    def take_damage(self, amount: int, dtype: str = 'normal',
                    pos: tuple = None):
        """Tìm section gần pos nhất rồi truyền damage xuống."""
        section = self._find_nearest_section(pos)
        if section is not None:
            section.take_damage(amount, dtype)
            if not section.is_alive:
                GameEventBus.get_instance().publish(
                    'wall_breached', {'wall': self, 'section': section}
                )

    def is_destroyed(self) -> bool:
        """True khi TẤT CẢ section đều bị phá."""
        return all(not s.is_alive for s in self._sections)

    def get_hp_percent(self) -> float:
        """Tổng HP còn lại / tổng HP tối đa."""
        total     = sum(s._hp     for s in self._sections)
        total_max = sum(s._max_hp for s in self._sections)
        return total / total_max if total_max > 0 else 0.0

    def repair_all(self, amount: int):
        """Sửa tất cả section (RepairStation gọi)."""
        for s in self._sections:
            s.repair(amount)

    def update(self, dt: float):
        for s in self._sections:
            s.update(dt)

    def draw(self, screen):
        for s in self._sections:
            s.draw(screen)

    def _find_nearest_section(self, pos: tuple):
        """Tìm section gần tọa độ pos nhất."""
        if pos is None:
            return self._sections[0] if self._sections else None
        px, py = pos
        alive  = [s for s in self._sections if s.is_alive]
        if not alive:
            return None
        return min(alive, key=lambda s: (s.x-px)**2 + (s.y-py)**2)

    """Singleton — kho tài nguyên duy nhất của game.

    Mọi class cần tài nguyên đều gọi ResourceManager.get_instance().
    Không ai được cộng/trừ tài nguyên trực tiếp ngoài class này.
    """

    _instance = None

    def __init__(self):
        self._stock = ResourceBundle()

    @classmethod
    def get_instance(cls) -> "ResourceManager":
        if cls._instance is None:
            cls._instance = ResourceManager()
        return cls._instance

    def earn(self, bundle: ResourceBundle):
        """Cộng tài nguyên vào kho (loot, harvest, thưởng)."""
        self._stock += bundle

    def spend(self, cost: ResourceBundle) -> bool:
        """Trừ tài nguyên. Trả về False nếu không đủ."""
        if not self.can_afford(cost):
            return False
        self._stock.wood            -= cost.wood
        self._stock.stone           -= cost.stone
        self._stock.gas             -= cost.gas
        self._stock.food            -= cost.food
        self._stock.ore             -= cost.ore
        self._stock.anti_armor_bolt -= cost.anti_armor_bolt
        return True

    def can_afford(self, cost: ResourceBundle) -> bool:
        """Kiểm tra đủ tài nguyên không."""
        return self._stock >= cost

    def get_stock(self) -> ResourceBundle:
        """Trả về bản sao kho hiện tại. HUD dùng để hiển thị."""
        return self._stock

    def apply_defeat_penalty(self):
        """Thua màn → mất 20% tài nguyên."""
        self._stock = self._stock * 0.8

class FrozenDecorator:
    """Titan bị đóng băng — giảm tốc độ."""

    def __init__(self, entity, slow_amount: float, duration: float):
        """
        Tham số:
            entity:     Titan cần bọc
            slow_amount: 0.4 = slow 40%
            duration:    giây hiệu ứng tồn tại
        """
        self._entity     = entity
        self._slow        = slow_amount
        self._remaining  = duration

        # Lưu speed gốc để phục hồi sau
        self._orig_speed = entity._speed
        entity._speed   = entity._speed * (1 - slow_amount)

        # Delegate các thuộc tính cần thiết
        self.x          = entity.x
        self.y          = entity.y
        self.id         = entity.id
        self.is_alive   = entity.is_alive

    def update(self, dt: float):
        """Đếm ngược. Khi hết → phục hồi speed gốc."""
        self._remaining -= dt
        if self._remaining <= 0:
            # Tháo Decorator
            self._entity._speed = self._orig_speed
            return self._entity   # trả về entity gốc
        self._entity.update(dt)
        self.x        = self._entity.x
        self.y        = self._entity.y
        self.is_alive = self._entity.is_alive
        return self   # vẫn còn hiệu ứng

    def take_damage(self, amount: int, dtype: str):
        """Delegate xuống entity gốc."""
        self._entity.take_damage(amount, dtype)
        self.is_alive = self._entity.is_alive

        elif skill_id == 'E': self._nape_lock()
        elif skill_id == 'R': self._captain_buff()
